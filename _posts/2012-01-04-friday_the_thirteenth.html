int isleap(int year)
{
    if(year%400==0||(year%4==0)&&year%100!=0&&!year%400==0)
        return 1;
    return 0;
}

int days(int year,int month,int day)
{
    int n=0,i;
    int mon[12];
    mon[0]=mon[2]=mon[4]=mon[6]=mon[7]=mon[9]=mon[11]=31;
    mon[3]=mon[5]=mon[8]=mon[10]=30;
    if(isleap(year))
        mon[1]=29;
    else
        mon[1]=28;
    if(day>mon[month-1]||day<1||month<1||month>12)
    {
        printf("error!");
        return 0;
    }
    for(i=0;i        n+=mon[i];
    n=n+day;
    return n;
}
int ndaysweek(int year,int month,int day)
{
    long i,n;
    n=days(year,month,day);
    for(i=year-1;i>=1900;i--)
    n+=days(i,12,31);
    return n%7;
}
